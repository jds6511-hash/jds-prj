# 영상 검색 웹 UI (M7-W) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 영상 업로드 → M1~M4 자동 인덱싱(진행률 표시) → 좌측 플레이어 + 우측 전사/장면설명 목록 + 채팅형 검색 웹 UI를 FastAPI + 단일 HTML로 구현한다.

**Architecture:** `src/m7_webui.py`(FastAPI 서버)가 M5의 `search()`/`VideoIndex`를 그대로 import하고, M1~M4는 기존 CLI를 서브프로세스로 순차 실행한다(파일 계약·멱등성 상속). 화면은 `src/webui/index.html` 단일 파일(빌드 도구 없음), 네이티브 `<video>` API로 시크. 스펙: `docs/superpowers/specs/2026-07-07-webui-design.md`.

**Tech Stack:** Python 3.12, FastAPI, uvicorn, python-multipart, httpx(테스트), 순수 HTML/CSS/JS

## Global Constraints (스펙 확정 규칙)

- 검색은 `m5_search.search()`/`VideoIndex`를 그대로 import — 재구현 금지
- 기존 코드(src/m1~m7, common, tests)는 수정 금지. 새 파일만 추가
- M1~M4 실행은 서브프로세스 CLI(`python src/mN_*.py --config {path} --video-id {id}`) — import 실행 금지
- 인덱싱은 서버 전체 동시 1건만. 진행 중 업로드는 409 + "다른 영상 인덱싱 중"
- video_id 정규화: 파일명 stem에서 `[^a-zA-Z0-9_-]`를 `_`로 치환
- α는 CLI `--alpha`(기본 0.5) — help에 "M6 grid search 후 고정값으로 교체하는 임시값" 명시
- mp4 아닌 업로드 400, 인덱싱 완료 전 검색 409, 미지 video_id status 404
- 검색 응답은 Top-3 고정: `{idx, start(int), end(int), score, subtitle, caption}`
- UI 시간 표시는 m:ss, 내부 값은 정수 초
- 클라우드 API 금지 — 전부 로컬(127.0.0.1) 서빙
- 테스트는 `python -m pytest tests/ -v`로 실행 (bare pytest 금지 — tests/ 밖에 GPU 로딩 파일 존재)

```python
    deadline = time.time() + timeout

        st = client.get(f"/api/status/{vid}").json()
```

```python
"""M7-W 웹 UI 서버: 업로드 → M1~M4 서브프로세스 인덱싱 → 채팅 검색.
검색은 m5_search.search를 그대로 import (재구현 금지).
[docs/superpowers/specs/2026-07-07-webui-design.md]"""
import argparse, re, subprocess, sys, threading
from pathlib import Path

from fastapi import FastAPI, HTTPException, UploadFile
from fastapi.responses import FileResponse, HTMLResponse

import common
from m5_search import VideoIndex, search

PIPELINE = ("m1_preprocess.py", "m2_keyframe.py", "m3_generate.py", "m4_index.py")
STAGE = {"m1_preprocess.py": "m1", "m2_keyframe.py": "m2",
         "m3_generate.py": "m3", "m4_index.py": "m4"}


def sanitize_video_id(stem: str) -> str:
    return re.sub(r"[^a-zA-Z0-9_-]", "_", stem)


def run_module_subprocess(script: str, config_path: str, video_id: str) -> None:
    """M1~M4 CLI 한 단계 실행. 실패 시 stderr 꼬리를 담아 RuntimeError."""
    proc = subprocess.run(
        [sys.executable, str(Path("src") / script),
         "--config", config_path, "--video-id", video_id],
        capture_output=True, text=True, encoding="utf-8", errors="replace")
    if proc.returncode != 0:
        tail = "\n".join((proc.stderr or "").strip().splitlines()[-5:])
        raise RuntimeError(f"{script} 실패:\n{tail}")


class JobStore:
    """인덱싱 상태 저장소. GPU 자원 보호를 위해 동시 1건만 허용."""
    def __init__(self):
        self._lock = threading.Lock()
        self._status: dict[str, dict] = {}
        self._busy = False

    def try_start(self, video_id: str) -> bool:
        with self._lock:
            if self._busy:
                return False
            self._busy = True
            self._status[video_id] = {"stage": "m1", "detail": "시작 대기"}
            return True

    def set(self, video_id: str, stage: str, detail: str = "") -> None:
        with self._lock:
            self._status[video_id] = {"stage": stage, "detail": detail}
            if stage in ("done", "error"):
                self._busy = False

    def get(self, video_id: str) -> dict | None:
        with self._lock:
            return self._status.get(video_id)


def create_app(cfg: dict, config_path: str, alpha: float,
               run_module=run_module_subprocess,
               search_fn=search, load_index=VideoIndex.load) -> FastAPI:
    app = FastAPI()
    jobs = JobStore()
    index_cache: dict[str, VideoIndex] = {}
    videos_dir = Path(cfg["paths"]["data"]) / "videos"
    html_path = Path(__file__).parent / "webui" / "index.html"

    def _pipeline(video_id: str) -> None:
        try:
            for script in PIPELINE:
                jobs.set(video_id, STAGE[script], f"{script} 실행 중")
                run_module(script, config_path, video_id)
            index_cache.pop(video_id, None)      # 재인덱싱 시 캐시 무효화
            jobs.set(video_id, "done")
        except Exception as e:                   # 단계 실패 → UI에 원인 표시
            jobs.set(video_id, "error", str(e))

    @app.post("/api/upload")
    async def upload(file: UploadFile):
        if not (file.filename or "").lower().endswith(".mp4"):
            raise HTTPException(400, "mp4 파일만 업로드할 수 있어요")
        video_id = sanitize_video_id(Path(file.filename).stem)
        if not jobs.try_start(video_id):
            raise HTTPException(409, "다른 영상 인덱싱 중이에요 — 잠시 후 다시 시도하세요")
        videos_dir.mkdir(parents=True, exist_ok=True)
        (videos_dir / f"{video_id}.mp4").write_bytes(await file.read())
        threading.Thread(target=_pipeline, args=(video_id,), daemon=True).start()
        return {"video_id": video_id}

    @app.get("/api/status/{video_id}")
    def status(video_id: str):
        st = jobs.get(video_id)
        if st is None:
            raise HTTPException(404, f"{video_id}: 업로드 기록 없음")
        return st

    return app


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--config", default="config.yaml")
    ap.add_argument("--alpha", type=float, default=0.5,
                    help="M6 grid search 후 고정값으로 교체하는 임시값")
    ap.add_argument("--port", type=int, default=7860)
    args = ap.parse_args()
    import uvicorn
    cfg = common.load_config(args.config)
    uvicorn.run(create_app(cfg, args.config, args.alpha),
                host="127.0.0.1", port=args.port)


if __name__ == "__main__":
    main()
```

```python
    @app.get("/api/segments/{video_id}")
    def segments(video_id: str):
        path = common.work_dir(cfg, video_id) / "segments.json"
        if not path.exists():
            raise HTTPException(404, f"{video_id}: 인덱스 없음")
        doc = common.load_segments(path, require=["subtitle", "caption"])
        keys = ("idx", "start", "end", "subtitle", "caption")
        return {"segments": [{k: s[k] for k in keys} for s in doc["segments"]]}

    @app.post("/api/search")
    def do_search(body: dict):
        video_id, query = body.get("video_id", ""), body.get("query", "")
        if not query.strip():
            raise HTTPException(400, "질의가 비어 있어요")
        st = jobs.get(video_id)
        if st is not None and st["stage"] != "done":
            raise HTTPException(409, "인덱싱이 끝나면 검색할 수 있어요")
        if video_id not in index_cache:
            try:
                index_cache[video_id] = load_index(cfg, video_id)
            except FileNotFoundError as e:       # 산출물 미존재 → 안내
                raise HTTPException(409, str(e))
        video = index_cache[video_id]
        top = search_fn(query, video, alpha, cfg)[:3]
        return {"results": [
            {"idx": r.idx, "start": int(r.start), "end": int(r.end),
             "score": round(r.score, 3),
             "subtitle": video.segments[r.idx]["subtitle"],
             "caption": video.segments[r.idx]["caption"]} for r in top]}

    @app.get("/api/video/{video_id}")
    def video_file(video_id: str):
        p = videos_dir / f"{sanitize_video_id(video_id)}.mp4"
        if not p.exists():
            raise HTTPException(404, "영상 파일 없음")
        return FileResponse(p, media_type="video/mp4")   # starlette Range 지원
```

```html
<script>
const $ = id => document.getElementById(id);
let videoId = null;
const STAGE_INFO = { m1: [10, "오디오 추출·세그먼트 분할 (M1)"],
                     m2: [30, "대표 프레임 선택 (M2)"],
                     m3: [60, "전사·장면설명 생성 (M3)"],
                     m4: [85, "임베딩 인덱싱 (M4)"] };

function fmt(sec) {
  sec = Math.floor(sec);
  return Math.floor(sec / 60) + ":" + String(sec % 60).padStart(2, "0");
}
function seek(sec) {
  const v = $("player");
  v.currentTime = sec;
  v.play();
}
function showError(text) {
  $("error-box").style.display = "block";
  $("error-box").textContent = text;
}

$("file-input").addEventListener("change", async e => {
  const file = e.target.files[0];
  if (!file) return;
  $("error-box").style.display = "none";
  const fd = new FormData();
  fd.append("file", file);
  const r = await fetch("/api/upload", { method: "POST", body: fd });
  const body = await r.json();
  if (!r.ok) { showError(body.detail); return; }
  videoId = body.video_id;
  $("progress-wrap").style.display = "flex";
  poll();
});

async function poll() {
  const r = await fetch(`/api/status/${videoId}`);
  const st = await r.json();
  if (st.stage === "error") {
    showError("인덱싱 실패\n" + st.detail);
    $("progress-wrap").style.display = "none";
    return;
  }
  if (st.stage === "done") {
    $("progress-fill").style.width = "100%";
    $("progress-text").textContent = "완료";
    setTimeout(() => { $("progress-wrap").style.display = "none"; }, 1500);
    onReady();
    return;
  }
  const [pct, label] = STAGE_INFO[st.stage] || [0, st.stage];
  $("progress-fill").style.width = pct + "%";
  $("progress-text").textContent = label + "…";
  setTimeout(poll, 1000);
}

async function onReady() {
  $("player").src = `/api/video/${videoId}`;
  const r = await fetch(`/api/segments/${videoId}`);
  const { segments } = await r.json();
  const box = $("segments");
  box.innerHTML = "";
  for (const s of segments) {
    const div = document.createElement("div");
    div.className = "seg";
    div.id = `seg-${s.idx}`;
    div.innerHTML = `<span class="t">${fmt(s.start)}</span>` +
      `<span>${s.subtitle || "(무발화)"}</span>` +
      `<div class="cap">${s.caption}</div>`;
    div.onclick = () => seek(s.start);
    box.appendChild(div);
  }
  $("chat-input").disabled = false;
  $("chat-send").disabled = false;
  addMsg("bot", null, "인덱싱 완료! 찾고 싶은 장면을 설명해 보세요.");
}

function addMsg(role, cards, text) {
  const div = document.createElement("div");
  div.className = `msg ${role}`;
  if (text) div.textContent = text;
  if (cards) for (const c of cards) {
    const card = document.createElement("div");
    card.className = "card";
    card.innerHTML = `<span class="time">${fmt(c.start)}~${fmt(c.end)}</span> ` +
      `<span>${c.subtitle || "(무발화)"}</span>` +
      `<div class="cap">${c.caption}</div>`;
    card.onclick = () => seek(c.start);
    div.appendChild(card);
  }
  $("chat-log").appendChild(div);
  $("chat-log").scrollTop = $("chat-log").scrollHeight;
}
```
